Algoritmia y Ordenamiento

Un algoritmo de ordenamiento organiza datos, pero en Python, la elección entre .sort() y sorted() define cómo interactúa tu programa con la memoria (RAM) y la estructura de los datos.
IN-PLACE
.sort()
MÉTODO EXCLUSIVO DE LISTAS

Reescribe los bloques de memoria originales destruyendo el orden anterior y retorna None.

  • ✓ Vía más rápida y eficiente
  • ✓ Ideal para procesar BD masivas
  • ✗ No clona información (destructivo)
NUEVA MEMORIA
sorted()
FUNCIÓN UNIVERSAL (ITERABLES)

Asigna nueva memoria, crea una lista desde cero, la ordena y te la entrega. Acepta tuplas, diccionarios, etc.

  • ✓ Protege la información original
  • ✓ Ideal para programación funcional
  • ✗ Exige más RAM temporal
⚡
Bajo el capó, ambos utilizan Timsort, un motor de ordenamiento híbrido ultrarrápido derivado de Merge Sort e Insertion Sort.
> COMPARATIVA: ANTES VS DESPUÉS
✗ CÓDIGO INEFICIENTE
burbuja.py
# Ordenamiento manual — confuso y verboso
# No se sabe qué criterio usa, ni si
# modifica la lista o crea una nueva.
datos = [5, 2, 8, 1, 9]

# ❌ Bubble sort a mano sin estructura
for i in range(len(datos)):
  for j in range(len(datos)):
    if datos[j] > datos[i]:
      tmp = datos[i]
      datos[i] = datos[j]
      datos[j] = tmp

# ❌ No sabemos si original fue modificado
print(datos)

# ❌ Para objetos complejos, no funciona
personas = [("Ana", 30), ("Luis", 25)]
for i in range(len(personas)):
  for j in range(len(personas)):
    if personas[j][1] > personas[i][1]:
      personas[i], personas[j] = \
        personas[j], personas[i]
✓ CÓDIGO OPTIMIZADO
sort_optimizado.py
# .sort() — in-place: modifica la lista
# sorted() — retorna nueva lista sin tocar
# la original. Timsort O(n log n).
datos = [5, 2, 8, 1, 9]

# ✅ sorted(): original intacta
nueva = sorted(datos)
print(datos)   # [5, 2, 8, 1, 9] — sin cambios
print(nueva)   # [1, 2, 5, 8, 9] — nueva lista

# ✅ .sort(): modifica in-place, + rápido
datos.sort(reverse=True)
print(datos)   # [9, 8, 5, 2, 1]

# ✅ Objetos complejos: key=lambda
personas = [("Ana", 30), ("Luis", 25)]
por_edad = sorted(
  personas,
  key=lambda p: p[1]  # ordena por edad
)
print(por_edad)
# [('Luis', 25), ('Ana', 30)]

bubble_sort.py terminal README
# ==========================================
# 1. BUBBLE SORT
# ==========================================
mi_lista = [5, 2, 9, 1]
n = len(mi_lista)
paso = 1
print("=== BUBBLE SORT EN ACCIÓN ===")
print(f"Lista a ordenar: {mi_lista}\n")
for i in range(n):
intercambiado = False
for j in range(n - 1 - i):
v1, v2 = mi_lista[j], mi_lista[j+1]
if v1 > v2:
# Si el primero es mayor, los intercambiamos
mi_lista[j], mi_lista[j+1] = mi_lista[j+1], mi_lista[j]
print(f"[Paso {paso}] Analizando [{v1}, {v2}]... ¡Se mueven! Lista actual: {mi_lista}")
intercambiado = True
else:
# Si ya están en orden, no hacemos nada
print(f"[Paso {paso}] Analizando [{v1}, {v2}]... Todo bien.")
paso += 1
# Resumen al final de cada recorrido
print(f"> Fin de la ronda {i + 1}. Progreso: {mi_lista}\n")
if not intercambiado:
break
print(f"=== RESULTADO FINAL ===\nLista ordenada: {mi_lista}\n")
print("=" * 60)
# ==========================================
# 2. EJEMPLO DE .sort()
# ==========================================
print("\n--- Analizando .sort() ---")
puntajes = [88, 45, 100, 33]
print(f"Original: {puntajes}")
resultado_metodo = puntajes.sort() # Ordenamos la lista directamente
print(f"Después de .sort(): {puntajes} (La lista original cambió)")
print(f"Valor devuelto: {resultado_metodo} (Por eso nunca debes hacer x = lista.sort())")
# ==========================================
# 3. EJEMPLO DE sorted()
# ==========================================
print("\n--- Analizando sorted() ---")
vocales = ('u', 'e', 'o', 'a', 'i') # Esto es una tupla, no tiene .sort()
print(f"Original: {vocales}")
# Usamos la función para generar una lista nueva
nueva_lista = sorted(vocales)
print(f"La nueva lista: {nueva_lista}")
print(f"La original intacta: {vocales}")

Gestión de Excepciones

Las excepciones no tratadas provocan el cierre forzoso de un programa. El bloque try / except / finally es el escudo protector que intercepta estos fallos antes de que destruyan la ejecución.
MONITOR
try:
ZONA DE PRUEBAS

Crea un entorno aislado para ejecutar el código. Evalúa las instrucciones de forma directa y no penaliza el rendimiento del procesador si todo sale bien.

INTERCEPTOR
except:
MANEJO DE ERRORES

Evita el colapso y el volcado de memoria (traceback). Captura el objeto de error específico (ej. ValueError) y desvía el flujo hacia una solución segura.

GARANTÍA
finally:
LIMPIEZA DE MEMORIA

CPython fuerza su ejecución incluso tras un error fatal o un return prematuro. Su rol es quitar candados a archivos y liberar recursos de la máquina.

> COMPARATIVA: ANTES VS DESPUÉS
✗ CÓDIGO INEFICIENTE

robusto.py
# try/except/finally: manejo elegante
# Mensajes claros, flujo controlado.

archivo = None
try:
    # ✅ Captura si no es número válido
    numero = int(input("Ingrese número: "))

    # ✅ Captura si el archivo no existe
    archivo = open("datos.txt")
    contenido = archivo.read()

    # ✅ Captura si divisor es cero
    divisor = int(input("Divisor: "))
    result = numero / divisor

except ValueError as e:
    print(f"⚠ ERROR: Valor inválido → {e}")
except FileNotFoundError:
    print("⚠ ERROR: Archivo no encontrado")
except ZeroDivisionError:
    print("⚠ ERROR: División por cero")
finally:
    # ✅ Siempre se ejecuta: limpieza segura
    if archivo: archivo.close()
    print("[finally] Recursos liberados.")

# ═══════════════════════════════════════════════════════
# Modulo 02 | Gestión de Excepciones en Terminal
# Python — Calculadora Segura
# ═══════════════════════════════════════════════════════
# Códigos ANSI para colores
ROJO = "\033[91m"; VERDE = "\033[92m"; MAGENTA = "\033[95m"; RESET = "\033[0m"
def calculadora_segura():
"""Una calculadora básica a prueba de fallos para principiantes."""
while True:
print(f"\n{VERDE}=== CALCULADORA BÁSICA PARA DIVISIONES ==={RESET}")
try:
# Solicitamos los datos al usuario
num1 = int(input("Ingresa el primer número: "))
num2 = int(input("Ingresa el segundo número: "))
# Intentamos hacer la operación matemática
resultado = num1 / num2
# Si llegamos aquí, NADA falló
print(f"{VERDE}✓ ÉXITO: El resultado de {num1} / {num2} es {resultado}{RESET}")
break # Rompemos el bucle infinito porque tuvimos éxito
except ValueError:
# Ocurre si escriben letras o símbolos en vez de números
print(f"{ROJO}✗ ERROR [ValueError]: Por favor, ingresa solo números.{RESET}")
except ZeroDivisionError:
# Ocurre si el segundo número (num2) es un cero
print(f"{ROJO}✗ ERROR [ZeroDivisionError]: No se puede dividir entre cero.{RESET}")
finally:
# Esto SIEMPRE se imprime, sin importar si hubo error o éxito
print(f"{MAGENTA}⚡ [finally] Limpiando memoria de la calculadora...{RESET}")
if __name__ == "__main__":
calculadora_segura()

F-Strings y Reportes Profesionales

Los f-strings (Formatted String Literals) revolucionaron la presentación de datos en Python, dejando obsoletos tanto a la concatenación tradicional (+) como al clásico .format().
VELOCIDAD
Evaluación Dinámica
COMPILACIÓN DIRECTA EN C

A diferencia de los métodos clásicos que buscan y reemplazan marcadores en múltiples pasos, los f-strings son evaluados en tiempo de ejecución. Python compila las variables de las llaves {} directamente en operaciones súper rápidas a bajo nivel.

EFICIENCIA
Gestión de Memoria
ASIGNACIÓN ÚNICA

En Python los textos son inmutables. Sumarlos con + genera copias basura constantes. El f-string en cambio precalcula el tamaño final exacto y pide la memoria a la RAM de un solo golpe, evitando cuellos de botella.

⚡
Por último, su Mini-Lenguaje de Formato (ej. :>9.2f) traslada tareas pesadas de procesamiento (como alinear, rellenar con ceros o redondear decimales) al motor interno.
> COMPARATIVA: ANTES VS DESPUÉS
✗ CÓDIGO INEFICIENTE

# f-strings con especificadores de formato
# Legible, preciso y potente.

precio = 1234.5678
nombre = "Laptop"

# ✅ Redondeo a 2 decimales
print(f"Precio: ${precio:.2f}")   # $1234.57

# ✅ Alineación izquierda 20 chars
print(f"{nombre:<20}| ok")      # "Laptop               | ok"

# ✅ Alineación derecha
print(f"{nombre:>20}")         # "              Laptop"

# ✅ Centrado
print(f"{nombre:^20}")         # "       Laptop       "

# ✅ Relleno con ceros
print(f"Ticket #{42:0>6}")       # Ticket #000042

# ✅ Separador de miles
print(f"Stock: {1000000:,}")     # Stock: 1,000,000

# ═══════════════════════════════════════════════════════
# MOD_03 | F-Strings y Reportes Profesionales
# Python 3.11+ — Formateo avanzado y generador de tickets
# ═══════════════════════════════════════════════════════
from datetime import datetime
# ── Datos del ticket ──────────────────────────────────
productos: list[dict] = [
{"nombre": "Manzanas", "cantidad": 3, "precio": 1.50},
{"nombre": "Pan integral", "cantidad": 1, "precio": 2.99},
{"nombre": "Leche 1L", "cantidad": 2, "precio": 1.20},
{"nombre": "Jugo naranja", "cantidad": 4, "precio": 0.99},
]
ANCHO = 44 # ancho total del ticket
IVA = 0.19 # 19%
NUM = 42 # número de ticket
def linea(inicio="╠", medio="═", fin="╣") -> str:
"""Genera una línea separadora de ancho fijo."""
return f"{inicio}{medio * (ANCHO - 2)}{fin}"
def fila(nombre: str, cant: int, precio: float) -> str:
"""Genera una fila perfectamente tabulada con f-strings."""
# Total: 3 + 20 + 5 + 5 + 7 + 4 = 44 chars exactos
return f"║ {nombre:<20}{cant:>5} ${precio:>7.2f} ║"
def titulo(texto: str) -> str:
"""Centra el texto en el ancho del ticket."""
interior = f"{texto:^{ANCHO - 2}}"
return f"║{interior}║"
def fila_total(etiqueta: str, monto: float) -> str:
"""Fila de totales con etiqueta y monto alineados."""
# Total: 3 + 28 + 1 + 8 + 4 = 44 chars exactos (Corregido)
return f"║ {etiqueta:<28}${monto:>8.2f} ║"
def generar_ticket(items: list[dict], numero: int):
"""Genera e imprime el ticket completo."""
fecha = datetime.now().strftime("%Y-%m-%d")
subtotal = sum(p["cantidad"] * p["precio"] for p in items)
iva = subtotal * IVA
total = subtotal + iva
print(f"╔{'═'*(ANCHO-2)}╗")
print(titulo(f"TICKET DE VENTA #{numero:0>6}"))
print(linea())
print(titulo("Producto Cant. Precio"))
print(linea())
for p in items:
print(fila(p["nombre"], p["cantidad"], p["precio"]))
print(linea())
print(fila_total("SUBTOTAL:", subtotal))
print(fila_total(f"IVA ({IVA*100:.0f}%):", iva))
print(linea())
print(fila_total("TOTAL:", total))
print(f"╚{'═'*(ANCHO-2)}╝")
print(f" Generado: {fecha} | Items: {len(items):,}")
if __name__ == "__main__":
generar_ticket(productos, NUM)
